    <script>
      (function () {
        var STORAGE_KEY = "pageMarginPreset";
        var STORAGE_WIDTH_KEY = "pageShellMaxWidthPx";
        var STORAGE_LOCK_KEY = "pageMarginCustomLocked";
        var SHELL = {
          tight: "mx-auto max-w-7xl px-2 sm:px-3",
          balanced: "mx-auto max-w-6xl px-3 sm:px-4",
          relaxed: "mx-auto max-w-5xl px-5 sm:px-8",
          custom: "mx-auto w-full px-3 sm:px-4",
        };
        var PRESET_WIDTH_HINT = { tight: 1280, balanced: 1152, relaxed: 1024 };
        var ABS_MIN_W = 400;
        var ABS_MAX_W = 2500;
        var handleL = null;
        var handleR = null;
        var dragState = null;
        var rafScheduled = false;
        var settingsMenuPinned = false;

        function pinSettingsMenuIfOpen() {
          if (settingsMenuPinned) return;
          var menu = document.getElementById("app-settings-menu");
          if (!menu || menu.classList.contains("hidden")) return;
          var r = menu.getBoundingClientRect();
          menu.dataset.pinned = "1";
          settingsMenuPinned = true;
          menu.style.position = "fixed";
          menu.style.left = r.left + "px";
          menu.style.top = r.top + "px";
          menu.style.width = r.width + "px";
          menu.style.right = "auto";
          menu.style.marginTop = "0";
          menu.style.zIndex = "10050";
        }

        function unpinSettingsMenu() {
          var menu = document.getElementById("app-settings-menu");
          if (!menu || menu.dataset.pinned !== "1") return;
          delete menu.dataset.pinned;
          settingsMenuPinned = false;
          menu.style.position = "";
          menu.style.left = "";
          menu.style.top = "";
          menu.style.width = "";
          menu.style.right = "";
          menu.style.marginTop = "";
          menu.style.zIndex = "";
        }

        function maxWidthForViewport() {
          try {
            var vw = window.innerWidth || document.documentElement.clientWidth || ABS_MAX_W;
            return Math.max(ABS_MIN_W, Math.min(ABS_MAX_W, Math.floor(vw)));
          } catch (e) {
            return ABS_MAX_W;
          }
        }

        function clampWidth(w) {
          var n = Math.round(Number(w));
          if (isNaN(n)) n = 1600;
          var cap = maxWidthForViewport();
          return Math.min(Math.max(n, ABS_MIN_W), cap, ABS_MAX_W);
        }

        function isCustomLocked() {
          try {
            return localStorage.getItem(STORAGE_LOCK_KEY) === "1";
          } catch (e) {
            return false;
          }
        }

        function setCustomLocked(locked) {
          try {
            if (locked) localStorage.setItem(STORAGE_LOCK_KEY, "1");
            else localStorage.removeItem(STORAGE_LOCK_KEY);
          } catch (e) {}
          syncCustomLockUi();
        }

        function syncCustomLockUi() {
          var sel = document.getElementById("page-margin-select");
          var btn = document.getElementById("page-margin-custom-lock");
          var panel = document.getElementById("page-margin-custom-panel");
          var openIc = document.querySelector(".page-margin-lock-icon--open");
          var closedIc = document.querySelector(".page-margin-lock-icon--closed");
          var locked = isCustomLocked();
          var isCustom = sel && sel.value === "custom";
          if (btn) {
            if (isCustom) btn.classList.remove("hidden");
            else btn.classList.add("hidden");
            btn.setAttribute("aria-pressed", locked ? "true" : "false");
            btn.title = locked ? "Unlock custom width (show slider and edge handles)" : "Lock custom width (hide slider; width stays within screen)";
            btn.setAttribute(
              "aria-label",
              locked ? "Unlock custom width" : "Lock custom width"
            );
          }
          if (openIc && closedIc) {
            if (!isCustom || !locked) {
              openIc.classList.remove("hidden");
              closedIc.classList.add("hidden");
            } else {
              openIc.classList.add("hidden");
              closedIc.classList.remove("hidden");
            }
          }
          if (panel && isCustom) {
            if (locked) panel.classList.add("hidden");
            else panel.classList.remove("hidden");
          }
          scheduleHandleUpdate();
        }

        function updateSliderViewportMax() {
          var slider = document.getElementById("page-margin-width-slider");
          if (!slider) return;
          var cap = maxWidthForViewport();
          slider.setAttribute("max", String(cap));
          slider.setAttribute("aria-valuemax", String(cap));
          var sel = document.getElementById("page-margin-select");
          if (!sel || sel.value !== "custom") return;
          var v = parseInt(slider.value, 10);
          if (!isNaN(v) && v > cap) {
            slider.value = String(cap);
            previewCustomWidth(cap);
          }
        }

        function getStoredWidth() {
          try {
            var w = parseInt(localStorage.getItem(STORAGE_WIDTH_KEY), 10);
            if (!isNaN(w)) return clampWidth(w);
          } catch (e) {}
          return 1600;
        }

        function applyShellWithOptionalMax(shell, widthPx) {
          var h = document.getElementById("page-shell-header");
          var m = document.getElementById("page-shell-main");
          var f = document.getElementById("page-shell-footer");
          if (h) h.className = shell + " flex items-center justify-between gap-4 py-4";
          if (m) m.className = shell + " py-8";
          if (f) f.className = shell + " py-6 text-xs text-slate-500 dark:text-slate-400";
          [h, m, f].forEach(function (el) {
            if (!el) return;
            if (widthPx != null) {
              el.style.maxWidth = widthPx + "px";
            } else {
              el.style.maxWidth = "";
            }
          });
        }

        function previewCustomWidth(w) {
          var cw = clampWidth(w);
          applyShellWithOptionalMax(SHELL.custom, cw);
          try {
            document.documentElement.setAttribute("data-page-margin", "custom");
            document.documentElement.setAttribute("data-page-shell-max-w", String(cw));
          } catch (e) {}
          var valSpan = document.getElementById("page-margin-width-value");
          if (valSpan) valSpan.textContent = cw + "px";
          var slider = document.getElementById("page-margin-width-slider");
          if (slider && document.activeElement !== slider) {
            slider.value = String(cw);
          }
          scheduleHandleUpdate();
        }

        function commitCustomWidth(w) {
          var cw = clampWidth(w);
          applyShellWithOptionalMax(SHELL.custom, cw);
          try {
            document.documentElement.setAttribute("data-page-margin", "custom");
            document.documentElement.setAttribute("data-page-shell-max-w", String(cw));
            localStorage.setItem(STORAGE_KEY, "custom");
            localStorage.setItem(STORAGE_WIDTH_KEY, String(cw));
          } catch (e) {}
          var valSpan = document.getElementById("page-margin-width-value");
          if (valSpan) valSpan.textContent = cw + "px";
          var slider = document.getElementById("page-margin-width-slider");
          if (slider) slider.value = String(cw);
          scheduleHandleUpdate();
          unpinSettingsMenu();
        }

        function applyPageMarginPreset(preset) {
          var shell = SHELL[preset] || SHELL.balanced;
          applyShellWithOptionalMax(shell, null);
          try {
            document.documentElement.setAttribute("data-page-margin", preset);
            document.documentElement.removeAttribute("data-page-shell-max-w");
          } catch (e) {}
          scheduleHandleUpdate();
          unpinSettingsMenu();
        }

        function scheduleHandleUpdate() {
          if (rafScheduled) return;
          rafScheduled = true;
          requestAnimationFrame(function () {
            rafScheduled = false;
            updateDragHandlePositions();
            toggleDragHandlesVisibility();
          });
        }

        function updateDragHandlePositions() {
          if (!handleL || !handleR) return;
          var main = document.getElementById("page-shell-main");
          if (!main) return;
          var r = main.getBoundingClientRect();
          var top = r.top;
          var hgt = Math.max(0, r.height);
          handleL.style.top = top + "px";
          handleL.style.height = hgt + "px";
          handleL.style.left = Math.max(0, r.left - 4) + "px";
          handleR.style.top = top + "px";
          handleR.style.height = hgt + "px";
          handleR.style.left = Math.max(0, r.right - 4) + "px";
        }

        function toggleDragHandlesVisibility() {
          var show =
            document.documentElement.getAttribute("data-page-margin") === "custom" &&
            !isCustomLocked() &&
            window.matchMedia("(min-width: 640px)").matches;
          var d = show ? "" : "none";
          if (handleL) handleL.style.display = d;
          if (handleR) handleR.style.display = d;
        }

        function getMainMaxWidthPx() {
          var m = document.getElementById("page-shell-main");
          if (!m || !m.style.maxWidth) return getStoredWidth();
          var n = parseFloat(m.style.maxWidth);
          return isNaN(n) ? getStoredWidth() : n;
        }

        function ensureDragHandles() {
          if (handleL && handleR) return;
          handleL = document.createElement("div");
          handleR = document.createElement("div");
          handleL.className = "page-shell-drag-handle page-shell-drag-handle--left";
          handleR.className = "page-shell-drag-handle page-shell-drag-handle--right";
          handleL.setAttribute("role", "separator");
          handleR.setAttribute("role", "separator");
          handleL.setAttribute("aria-orientation", "vertical");
          handleR.setAttribute("aria-orientation", "vertical");
          handleL.setAttribute("aria-label", "Drag to resize content width");
          handleR.setAttribute("aria-label", "Drag to resize content width");
          handleL.addEventListener("pointerdown", onHandlePointerDown("left"));
          handleR.addEventListener("pointerdown", onHandlePointerDown("right"));
          document.body.appendChild(handleL);
          document.body.appendChild(handleR);
        }

        function onHandlePointerDown(side) {
          return function (e) {
            if (document.documentElement.getAttribute("data-page-margin") !== "custom") return;
            if (isCustomLocked()) return;
            e.preventDefault();
            dragState = {
              side: side,
              startX: e.clientX,
              startW: getMainMaxWidthPx(),
            };
            pinSettingsMenuIfOpen();
            (side === "left" ? handleL : handleR).classList.add("is-dragging");
            try {
              (side === "left" ? handleL : handleR).setPointerCapture(e.pointerId);
            } catch (err) {}
            document.addEventListener("pointermove", onHandlePointerMove);
            document.addEventListener("pointerup", onHandlePointerUp);
            document.addEventListener("pointercancel", onHandlePointerUp);
          };
        }

        function onHandlePointerMove(e) {
          if (!dragState) return;
          var dx = e.clientX - dragState.startX;
          var newW =
            dragState.side === "right"
              ? clampWidth(dragState.startW + 2 * dx)
              : clampWidth(dragState.startW - 2 * dx);
          dragState.lastW = newW;
          previewCustomWidth(newW);
        }

        function onHandlePointerUp(e) {
          if (!dragState) return;
          var side = dragState.side;
          var ds = dragState;
          dragState = null;
          var finalW = ds.lastW != null ? ds.lastW : ds.startW;
          if (handleL) handleL.classList.remove("is-dragging");
          if (handleR) handleR.classList.remove("is-dragging");
          try {
            var el = side === "left" ? handleL : handleR;
            if (el && el.releasePointerCapture) el.releasePointerCapture(e.pointerId);
          } catch (err) {}
          document.removeEventListener("pointermove", onHandlePointerMove);
          document.removeEventListener("pointerup", onHandlePointerUp);
          document.removeEventListener("pointercancel", onHandlePointerUp);
          commitCustomWidth(finalW);
        }

        function initPageMargin() {
          var boot = document.getElementById("page-shell-boot");
          if (boot) boot.remove();

          var sel = document.getElementById("page-margin-select");
          var slider = document.getElementById("page-margin-width-slider");
          var customPanel = document.getElementById("page-margin-custom-panel");
          var valSpan = document.getElementById("page-margin-width-value");
          var saved = null;
          try {
            saved = localStorage.getItem(STORAGE_KEY);
          } catch (e) {}

          ensureDragHandles();

          if (saved === "custom") {
            if (sel) sel.value = "custom";
            updateSliderViewportMax();
            var w0 = clampWidth(getStoredWidth());
            if (slider) slider.value = String(w0);
            if (valSpan) valSpan.textContent = w0 + "px";
            if (customPanel && !isCustomLocked()) customPanel.classList.remove("hidden");
            else if (customPanel && isCustomLocked()) customPanel.classList.add("hidden");
            commitCustomWidth(w0);
          } else if (saved && SHELL[saved]) {
            if (sel) sel.value = saved;
            applyPageMarginPreset(saved);
          } else if (sel) {
            applyPageMarginPreset(sel.value);
          }

          updateSliderViewportMax();
          syncCustomLockUi();

          var lockBtn = document.getElementById("page-margin-custom-lock");
          if (lockBtn) {
            lockBtn.addEventListener("click", function (e) {
              e.stopPropagation();
              if (!sel || sel.value !== "custom") return;
              if (isCustomLocked()) {
                setCustomLocked(false);
              } else {
                var w = clampWidth(getMainMaxWidthPx());
                commitCustomWidth(w);
                setCustomLocked(true);
              }
            });
          }

          if (sel) {
            var prevPreset = sel.value;
            sel.addEventListener("change", function () {
              var v = sel.value;
              if (v === "custom") {
                updateSliderViewportMax();
                var rawW = parseInt(localStorage.getItem(STORAGE_WIDTH_KEY), 10);
                var w = !isNaN(rawW) ? clampWidth(rawW) : clampWidth(PRESET_WIDTH_HINT[prevPreset] || 1152);
                if (customPanel && !isCustomLocked()) customPanel.classList.remove("hidden");
                else if (customPanel && isCustomLocked()) customPanel.classList.add("hidden");
                if (slider) slider.value = String(w);
                if (valSpan) valSpan.textContent = w + "px";
                commitCustomWidth(w);
              } else {
                if (customPanel) customPanel.classList.add("hidden");
                applyPageMarginPreset(v);
                try {
                  localStorage.setItem(STORAGE_KEY, v);
                  localStorage.removeItem(STORAGE_WIDTH_KEY);
                  localStorage.removeItem(STORAGE_LOCK_KEY);
                } catch (e) {}
              }
              prevPreset = v;
              syncCustomLockUi();
            });
          }

          if (slider) {
            slider.addEventListener("pointerdown", function () {
              if (isCustomLocked()) return;
              if (sel && sel.value === "custom") pinSettingsMenuIfOpen();
            });
            slider.addEventListener("input", function () {
              if (isCustomLocked()) return;
              if (sel && sel.value !== "custom") return;
              var w = clampWidth(parseInt(slider.value, 10));
              previewCustomWidth(w);
            });
            slider.addEventListener("change", function () {
              if (isCustomLocked()) return;
              if (sel && sel.value !== "custom") return;
              var w = clampWidth(parseInt(slider.value, 10));
              commitCustomWidth(w);
            });
            slider.addEventListener("pointerup", function () {
              setTimeout(unpinSettingsMenu, 0);
            });
            slider.addEventListener("pointercancel", function () {
              setTimeout(unpinSettingsMenu, 0);
            });
            slider.addEventListener("blur", function () {
              if (isCustomLocked()) return;
              if (sel && sel.value !== "custom") return;
              commitCustomWidth(clampWidth(parseInt(slider.value, 10)));
            });
          }

          window.addEventListener("resize", function () {
            unpinSettingsMenu();
            updateSliderViewportMax();
            if (
              document.documentElement.getAttribute("data-page-margin") === "custom" &&
              isCustomLocked()
            ) {
              commitCustomWidth(clampWidth(getMainMaxWidthPx()));
            }
            scheduleHandleUpdate();
          });
          window.addEventListener("scroll", scheduleHandleUpdate, true);
          try {
            var mqWide = window.matchMedia("(min-width: 640px)");
            if (mqWide.addEventListener) {
              mqWide.addEventListener("change", scheduleHandleUpdate);
            } else if (mqWide.addListener) {
              mqWide.addListener(scheduleHandleUpdate);
            }
          } catch (eMq) {}
          scheduleHandleUpdate();
        }

        var menuBtn = document.getElementById("app-settings-menu-btn");
        var menu = document.getElementById("app-settings-menu");
        var menuRoot = document.getElementById("app-settings-dropdown");
        var personQuickAddBtn = document.getElementById("person-quick-add-btn");
        var personQuickAddMenu = document.getElementById("person-quick-add-menu");

        function closeSettingsMenu() {
          if (!menu || !menuBtn) return;
          unpinSettingsMenu();
          menu.classList.add("hidden");
          menuBtn.setAttribute("aria-expanded", "false");
        }

          if (!menu || !menuBtn) return;
          closePersonQuickAddMenu();
          menu.classList.remove("hidden");
          menuBtn.setAttribute("aria-expanded", "true");
        }

        function toggleSettingsMenu() {
          if (!menu || !menuBtn) return;
          if (menu.classList.contains("hidden")) openSettingsMenu();
          else closeSettingsMenu();
        }

        if (menuBtn && menu) {
          menuBtn.addEventListener("click", function (e) {
            e.stopPropagation();

        if (personQuickAddBtn && personQuickAddMenu) {

            togglePersonQuickAddMenu();
          });
        }

        document.addEventListener("click", function (e) {
          if (menuRoot && menu && menuRoot.contains(e.target)) return;
          if (personQuickAddRoot && personQuickAddMenu && personQuickAddRoot.contains(e.target)) return;
          closeSettingsMenu();
          closePersonQuickAddMenu();
        });

        document.addEventListener("keydown", function (e) {
          if (e.key === "Escape") {
            closeSettingsMenu();
            closePersonQuickAddMenu();
          }
        });

        var THEME_KEY = "theme";
        function applyTheme(value) {
          var root = document.documentElement;
          root.classList.remove(
            "dark",
            "theme-light",
            "theme-dark",
            "theme-purple",
            "theme-pink",
            "theme-blossom",
            "theme-city",
            "theme-jungle"
          );
          if (value === "purple") {
            root.classList.add("dark", "theme-purple");
          } else if (value === "pink") {
            root.classList.add("dark", "theme-pink");
          } else if (value === "blossom") {
            root.classList.add("dark", "theme-blossom");
          } else if (value === "city") {
            root.classList.add("dark", "theme-city");
          } else if (value === "jungle") {
            root.classList.add("dark", "theme-jungle");
          } else if (value === "light") {
            root.classList.add("theme-light");
          } else {
            root.classList.add("dark", "theme-dark");
          }
          try {
            localStorage.setItem(THEME_KEY, value);
          } catch (err) {}
        }

        function currentThemeFromDom() {
          var root = document.documentElement;
          if (root.classList.contains("theme-purple")) return "purple";
          if (root.classList.contains("theme-pink")) return "pink";
          if (root.classList.contains("theme-blossom")) return "blossom";
          if (root.classList.contains("theme-city")) return "city";
          if (root.classList.contains("theme-light")) return "light";
          if (root.classList.contains("dark")) return "dark";
          return "light";
        }

        var themeSel = document.getElementById("theme-select");
        if (themeSel) {
          var saved = null;
          try {
            saved = localStorage.getItem(THEME_KEY);
          } catch (e) {}
          if (saved && ["light", "dark", "purple", "pink", "blossom", "city", "jungle"].indexOf(saved) !== -1) {
            themeSel.value = saved;
          } else {
            themeSel.value = currentThemeFromDom();
          }
          themeSel.addEventListener("change", function (e) {
            e.stopPropagation();
            applyTheme(themeSel.value);
          });
          themeSel.addEventListener("click", function (e) {
            e.stopPropagation();
          });
        }

        initPageMargin();
      })();

      (function () {
        var SCROLL_PREFIX = "appScrollRestore:";
        function storageKey() {
          return SCROLL_PREFIX + location.pathname + location.search;
        }
        function saveScroll() {
          try {
            var y = window.scrollY || window.pageYOffset || 0;
            sessionStorage.setItem(storageKey(), String(y));
          } catch (e) {}
        }
        function restoreScroll() {
          try {
            /* Invoice cross-ref: row-note redirect uses #invoice-crossref-row-N — do not override with stored Y */
            if (/^#invoice-crossref-row-\d+$/.test(location.hash)) {
              try {
                sessionStorage.removeItem(storageKey());
              } catch (e2) {}
              return;
            }
            var raw = sessionStorage.getItem(storageKey());
            if (raw == null) return;
            sessionStorage.removeItem(storageKey());
            var y = parseInt(raw, 10);
            if (Number.isNaN(y) || y < 0) return;
            if (window.history && "scrollRestoration" in window.history) {
              window.history.scrollRestoration = "manual";
            }
            requestAnimationFrame(function () {
              requestAnimationFrame(function () {
                window.scrollTo(0, y);
              });
            });
          } catch (e) {}
        }
        window.addEventListener("beforeunload", saveScroll);
        window.addEventListener("pagehide", saveScroll);
        document.addEventListener("visibilitychange", function () {
          if (document.visibilityState === "hidden") saveScroll();
        });
        document.addEventListener(
          "click",
          function (e) {
            var a = e.target.closest("a[href]");
            if (!a) return;
            var href = (a.getAttribute("href") || "").trim();
            if (!href || href.charAt(0) === "#" || href.indexOf("javascript:") === 0) return;
            if (a.target === "_blank" || a.hasAttribute("download")) return;
            try {
              var u = new URL(href, location.href);
              if (u.origin !== location.origin) return;
              saveScroll();
            } catch (err) {}
          },
          true
        );
        document.addEventListener(
          "submit",
          function (e) {
            var form = e.target;
            if (!form || form.tagName !== "FORM") return;
            if (form.getAttribute("target") === "_blank") return;
            try {
              var action = (form.getAttribute("action") || "").trim();
              if (!action || action.charAt(0) === "#") return;
              var u = new URL(action, location.href);
              if (u.origin !== location.origin) return;
              saveScroll();
            } catch (err) {}
          },
          true
        );
        if (document.readyState === "loading") {
          document.addEventListener("DOMContentLoaded", restoreScroll);
        } else {
          restoreScroll();
        }
      })();
